(() => {
    let m1 = Array(100).fill(0).map(() => {
	    return Math.floor(Math.random() * 10);
    });
    
    let m2 = Array(100).fill(0).map(() => {
        return Math.floor(Math.random() * 10) + 10;
    });

    let results = m2.map((value, i) => {
        return value - m1[i];
    });

    const convert = (values) => {
        return values.map((i) => {
            return i.toString(16).padStart(4, '0');
        }).join('\n')
    }

    console.log(convert(m1.concat(m2)));
    console.log(results.join('\n'));
})();
